export default function handler(req, res) {
  const name = req.query.name || "baby";
  res.status(200).json({
    comment: `Keep going ${name}, you're doing amazing.`
  });
}
export default function handler(req, res) {
  const { lat = "0", lng = "0" } = req.query;
  res.status(200).json({
    lat: Number(lat),
    lng: Number(lng)
  });
}
